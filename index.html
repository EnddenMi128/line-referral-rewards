const SHEET_NAME = "工作表1"; function doPost(e) { try { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const name = e.parameter.name; const phone = e.parameter.phone; const ref = e.parameter.ref; const userId = e.parameter.userId; sheet.appendRow([ new Date(), name, phone, ref, userId ]); return ContentService .createTextOutput("OK") .setMimeType(ContentService.MimeType.TEXT); } catch (err) { return ContentService .createTextOutput("ERROR: " + err) .setMimeType(ContentService.MimeType.TEXT); } }
